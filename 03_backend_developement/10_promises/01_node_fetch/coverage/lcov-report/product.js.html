  return fetch(
    `https://decath-product-api.herokuapp.com/products/${product}`,
    {method: "GET"}
  )
    .then((response) => response.json())
    .then((result) => {
      return Promise.all(
        [
          result,
          searchBrands(result)
        ]
      );
    })

      console.log(results[0], results[1]);

    })
    .catch((error) => {
      console.warn(error);
    });
}
